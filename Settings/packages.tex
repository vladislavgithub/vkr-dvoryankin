% Подписи рисунков и таблиц по требованию кафедры АСУ:
% «Рисунок 1 — Описание», разделитель — длинное тире.
\usepackage[labelsep=endash]{caption}
% Polyglossia ставит \figurename = «Рис.» (с точкой), нам нужно
% «Рисунок». \AtBeginDocument не работает — polyglossia сбрасывает
% после нас. Используем \addto\captionsrussian — стандартный
% механизм babel/polyglossia, выполняется при выборе языка.
% Polyglossia при активации русского пересоздаёт caption-настройки.
% Кладём ВСЁ в \captionsrussian — переопределение имён И всех
% captionsetup. Тогда последним применяется наш labelsep=endash,
% и подпись становится «Таблица 1.1 — Название» как требует кафедра.
\addto\captionsrussian{%
    \renewcommand{\figurename}{Рисунок}%
    \renewcommand{\tablename}{Таблица}%
    \captionsetup{labelsep=endash}%
    \captionsetup[table]    {labelsep=endash, justification=raggedleft, singlelinecheck=off, position=above}%
    \captionsetup[figure]   {labelsep=endash, justification=centering,  singlelinecheck=on,  position=below}%
    \captionsetup[lstlisting]{labelsep=endash, justification=centering, singlelinecheck=on,  position=below}%
}
