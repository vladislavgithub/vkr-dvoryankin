% Отступы списков по требованию кафедры:
% «Положение метки 1.25 см, положение текста 0» — метка-нумерация
% стоит как абзацный отступ, тело пункта (включая продолжающиеся
% строки) начинается с левого края страницы, как обычный абзац.
% Это эквивалент Word: «Отступ слева 0, отступ первой строки 1.25 см».
% В enumitem прямой синоним этого — опция wide=<dimen>: пункт
% оформляется как обычный параграф с отступом первой строки.
% Между пунктами — без дополнительного интервала; межстрочный 1.5
% наследуется от \linespread{1.5}.
\setlist[enumerate,1]{
    wide=1.25cm,
    label=\arabic*.,
    labelsep=0.5em,
    topsep=0pt, parsep=0pt, partopsep=0pt, itemsep=0pt,
}
\setlist[itemize,1]{
    wide=1.25cm,
    label=\textendash,
    labelsep=0.5em,
    topsep=0pt, parsep=0pt, partopsep=0pt, itemsep=0pt,
}
\setlist[itemize,2]{label=$\circ$, leftmargin=1cm}



% Каждая новая строка должна начинаться с отступа равного 1.25 сантиметра.
\setlength{\parindent}{1.25cm} % отступ для абзаца

% Оглавление по образцу кафедры:
%   ВВЕДЕНИЕ, ГЛАВА N. НАЗВАНИЕ, ЗАКЛЮЧЕНИЕ, СПИСОК ЛИТЕРАТУРЫ,
%   ПРИЛОЖЕНИЕ N — все заглавными, без отступа.
%   Подразделы (1.1, 1.2, ...) — обычным регистром, с отступом.
% \dottedcontents{<section>}[<left-indent>]{<format>}{<label-width>}{<leader-sep>}
%   label-width — ширина области под метку («ГЛАВА N.»); должна
%     быть достаточной, иначе номер налезет на текст.
%   leader-sep — расстояние между точками дотлидера; меньшее значение
%     → точки гуще.
% Используем \titlecontents (titletoc): два формата на section ---
% для нумерованных глав (\section{...}) и безномерных (\section*{...}).
% Это решает проблему с label-width 9em, который смещал безномерные
% строки (ВВЕДЕНИЕ, ЗАКЛЮЧЕНИЕ, ПРИЛОЖЕНИЕ N) за левый край страницы.
% В оглавлении все строки 14 pt (как и основной текст);
% жирным — только записи section (главы и Введение/Заключение/...).
\titlecontents{section}[0pt]
    {\fontsize{14pt}{17pt}\selectfont}                % above-code
    {\bfseries\thecontentslabel.\quad}                % numbered: «ГЛАВА N. »
    {\bfseries}                                       % numberless
    {\bfseries\titlerule*[6pt]{.}\contentspage}       % filler + page
\titlecontents{subsection}[1.25cm]
    {\fontsize{14pt}{17pt}\selectfont}
    {\thecontentslabel.\quad}
    {}
    {\titlerule*[6pt]{.}\contentspage}

\renewcommand{\thesection}{ГЛАВА \arabic{section}}
\renewcommand{\thesubsection}{\arabic{section}.\arabic{subsection}}

% Размеры и расположение заголовков по требованию кафедры:
%   - Главы (\section): 16 pt, ЖИРНЫЙ ПРОПИСНЫМИ
%   - Подразделы (\subsection): 14 pt, жирным
%   - Безномерные (\section* — Введение, Заключение): 14 pt, жирным
% Все заголовки выравниваются по левому краю с абзацным отступом
% 1.25 см. Интервалы перед/после заголовков — минимальные
% (как обычный абзац), без «Word-style» дополнительных промежутков.
\usepackage{titlesec}
% Заголовки в стиле Word-шаблона кафедры:
%   - shape = block: первая строка с абзацным отступом 1.25 см,
%     перенос продолжающихся строк к левому краю (положение 0),
%     а не под букву после метки (как было бы при hang).
%   - выравнивание по ширине (без \raggedright) — длинные заголовки
%     растягиваются по странице, как обычный абзац.
%   - точка в конце добавлена через body-фильтр (#1 — тело заголовка),
%     а не через after-code: после-код в [block]-режиме отделяется
%     от тела абзацным разрывом и точка попадает на отдельную строку.
\newcommand{\eqSectionFormat}[1]{\MakeUppercase{#1}.}
\newcommand{\eqSubsectionFormat}[1]{#1.}
% Межстрочный 1.5 наследуется от документа (\linespread{1.5}) — внутри
% многострочного заголовка строки разнесены так же, как в основном тексте.
% Чтобы при этом не возвращался лишний пробел между заголовком и следующим
% абзацем, в \titlespacing* ниже задано отрицательное after-значение
% -0.5\baselineskip — оно компенсирует «висящую» нижнюю часть высокого
% заголовка, оставляя стык с body-текстом плотным.
\titleformat{\section}[block]
    {\fontsize{16pt}{19pt}\bfseries\selectfont}
    {\hspace*{1.25cm}\thesection.}{0.5em}{\eqSectionFormat}
% Введение, Заключение — без метки и без точки на конце (по образцу
% кафедры АСУ заголовки уровня section без номера остаются «голыми»).
\titleformat{name=\section,numberless}[block]
    {\fontsize{14pt}{17pt}\bfseries\selectfont}
    {\hspace*{1.25cm}}{0pt}{\MakeUppercase}
\titleformat{\subsection}[block]
    {\fontsize{14pt}{17pt}\bfseries\selectfont}
    {\hspace*{1.25cm}\thesubsection.}{0.5em}{\eqSubsectionFormat}
% Подразделы 1.X.Y оформлены тем же стилем, что и 1.X --- bold,
% межстрочный 1.5, без отступа от тела абзаца. В оглавление
% не включаются: tocdepth ограничен уровнем subsection.
\titleformat{\subsubsection}[block]
    {\fontsize{14pt}{17pt}\bfseries\selectfont}
    {\hspace*{1.25cm}\thesubsubsection.}{0.5em}{\eqSubsectionFormat}
\setcounter{tocdepth}{2}
\setcounter{secnumdepth}{3}
% \titlespacing*{<команда>}{<отступ слева>}{<до>}{<после>}
% Все заголовки — левый край 0pt; абзацный отступ 1.25 см для метки
% («ГЛАВА N.», «N.M.») задаётся через \hspace*{1.25cm} в \titleformat.
% After-значение 0pt — заголовок и body разделены только естественным
% межстрочным 1.5 (наследуется от документа). Отрицательная
% компенсация удалена: с -0.5\baselineskip body-параграф наезжал
% на подзаголовок.
\titlespacing*{\section}      {0pt}{0pt}{0pt}
\titlespacing*{name=\section,numberless}{0pt}{0pt}{0pt}
\titlespacing*{\subsection}   {0pt}{0pt}{0pt}
\titlespacing*{\subsubsection}{0pt}{0pt}{0pt}
